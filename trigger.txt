create or replace trigger ftrigger(no in varchar(10),type in varchar(10))
after update on flight
for each row
declare 
	cap number;
begin
	if type=='Economy' then
	select capacity_Economy into cap from flight where 			no=flightno;
	if cap ==0 then
	RAISE_APPLICATION_ERROR(-20500,'Not available');
	end if;
	else 
	select capacity_Business into cap from flight where 		no=flightno;
	if cap ==0 then
	RAISE_APPLICATION_ERROR(-20500,'Not available');
	end if;
end if;
end;
/
